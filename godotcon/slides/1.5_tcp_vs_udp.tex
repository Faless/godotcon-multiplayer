\begin{frame}[t]{TCP vs UDP}

\vspace{-1cm}

\begin{columns}[t]
  \begin{column}{.5\textwidth}
    \begin{block}{}
    \large{TCP}

    \begin{itemize}
        \item Connection based
        \item Reliable delivery
        \item Ordered delivery
        \item High latency
    \end{itemize}
    \end{block}
\end{column}
  \begin{column}{.5\textwidth}
    \begin{block}{}
    \large{UDP}
    \begin{itemize}
        \item Connectionless
        \item Unreliable delivery
        \item Unordered delivery
        \item Low latency
    \end{itemize}
    \end{block}
  \end{column}
\end{columns}

\end{frame}
